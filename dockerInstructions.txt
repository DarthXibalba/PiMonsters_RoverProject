# Start docker with local files available
$ sudo docker run -it -v \
	~/_source/PiMonsters_RoverProject/SightCode/ImageDataset:/ImageDataset \
	gcr.io/tensorflow/tensorflow:latest-devel


# Change into the virtual root directory to see all files
$ cd /

--how_many_training_steps 500 \

--output_graph = /ImageDataset/retrained_graph.pb \
--output_labels= /ImageDataset/retrained_labels.txt \
--image_dir /ImageDataset/Train
