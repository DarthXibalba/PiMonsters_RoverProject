# Start docker with local files available
$ sudo docker run -it -v \
	~/_source/PiMonsters_RoverProject/SightCode/ImageDataset:/ImageDataset \
	gcr.io/tensorflow/tensorflow:latest-devel


# Pull the latest code for TF
$ cd /tensorflow/
$ git pull


# Run the trainer using our input data
(In Docker)
python /tensorflow/tensorflow/examples/image_retraining/retrain.py \
--bottleneck_dir=/ImageDataset/bottlenecks \
--how_many_training_steps 500 \
--model_dir=/ImageDataset/inception \
--output_graph=/ImageDataset/retrained_graph.pb \
--output_labels=/ImageDataset/retrained_labels.txt \
--image_dir /ImageDataset/Train





### TEST THAT TENSORFLOW IS WORKING

sudo docker run -it -v \
	~/_source/tf_files:/tf_files \

python tensorflow/examples/image_retraining/retrain.py \
